103 def gibberish_token?(token, location: false)
104   token_letters = token.gsub(/[^a-zA-Z]/, "")
105   return false if token_letters.length < TITLE_MIN_LENGTH
106   return false if token_letters == token_letters.upcase
107 
108   token_vowels = token_letters.scan(/[aeiouy]/i).length
109   return true if token_vowels.zero?
110 
111   vowel_ratio = token_vowels.to_f / token_letters.length
112 
113   if token_letters.length >= LONG_TOKEN_MIN_LENGTH && vowel_ratio < LONG_TOKEN_MIN_VOWEL_RATIO
114     return true
115   end
116 
117   if token_letters.length >= 6 && token.match?(CONSONANT_CLUSTER_PATTERN)
118     return false if location && token_vowels >= 2
119 
120     return true
121   end
122 
123   if token_letters.length >= 5 && token.match?(TOKEN_CONSONANT_CLUSTER_PATTERN) && vowel_ratio < 0.35
124     return false if location && token_vowels >= 2
125 
126     return true
127   end

134 def lacks_common_bigrams?(token_letters)
135   return false if token_letters.length < BIGRAM_CHECK_MIN_LENGTH
136 
137   lower = token_letters.downcase
138   bigrams = (0...(lower.length - 1)).map { |index| lower[index, 2] }.select { |pair| pair.match?(/\A[a-z]{2}\z/) }
139   return false if bigrams.empty?
140 
141   unique_common = bigrams.uniq.count { |pair| COMMON_BIGRAMS.include?(pair) }
142   required = [ 3, token_letters.length / 4 ].max
143 
144   unique_common < required
145 end
