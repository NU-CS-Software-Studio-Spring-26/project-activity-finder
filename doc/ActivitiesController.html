18 def index
19   @city_query     = params[:city].to_s.strip
20   @title_query    = params[:q].to_s.strip
21   @category_query = params[:category].to_s.strip
22 
23   base_scope = Activity.publicly_visible.order(event_date: :asc)
24   if @city_query.present?
25     pattern = "%#{ActiveRecord::Base.sanitize_sql_like(@city_query)}%"
26     base_scope = base_scope.where("city ILIKE ?", pattern)
27   end
28 
29   # Layout is an explicit, URL-persisted choice so it stays consistent across
30   # navigation (clicking a card and coming back keeps the same layout).
31   #   view=grouped → Airbnb-style horizontal rows grouped by category
32   #   view=list    → flat, paginated grid of every matching activity
33   # Some actions only make sense as a flat list, so they force list mode
34   # regardless of the toggle: a text search, a single-category filter,
35   # pagination, or a JSON request.
36   requested_view  = params[:view].to_s
37   inherently_list = @title_query.present? || @category_query.present? ||
38                     params[:page].present? || params[:per_page].present? ||
39                     request.format.json?
40   @browsing_mode  = !inherently_list && requested_view != "list"
41   @view_mode      = @browsing_mode ? "grouped" : "list"
42 
43   if @browsing_mode
44     all = base_scope.limit(300).to_a
45     ordered_cats = Activity::CATEGORIES + (all.map(&:category).uniq - Activity::CATEGORIES)
46     @activities_by_category = ordered_cats.filter_map do |cat|
47       acts = all.select { |a| a.category == cat }.first(12)
48       [ cat, acts ] unless acts.empty?
49     end.to_h
50 
51     all_ids = @activities_by_category.values.flatten.map(&:id)
52     @signup_counts       = ActivitySignup.where(activity_id: all_ids).group(:activity_id).count
53     @joined_activity_ids = ActivitySignup.where(user: current_user, activity_id: all_ids).pluck(:activity_id).to_set
54     @pagination = nil
55   else
56     if @title_query.present?
57       pattern = "%#{ActiveRecord::Base.sanitize_sql_like(@title_query)}%"
58       base_scope = base_scope.where(
59         "title ILIKE :q OR COALESCE(description, '') ILIKE :q OR COALESCE(category, '') ILIKE :q",
60         q: pattern
61       )
62     end
63     base_scope = base_scope.where(category: @category_query) if @category_query.present?
64 
65     result = paginate_activity_scope(base_scope, page_param: :page)
66     @activities = result[:records]
67     @pagination = result[:pagination]
68 
69     activity_ids = @activities.map(&:id)
70     @signup_counts       = ActivitySignup.where(activity_id: activity_ids).group(:activity_id).count
71     @joined_activity_ids = ActivitySignup.where(user: current_user, activity_id: activity_ids).pluck(:activity_id).to_set
72   end
73 end

110 def join
111   if @activity.user_id == current_user.id
112     redirect_to @activity, alert: "You’re hosting this activity."
113     return
114   end
115 
116   signup = @activity.activity_signups.find_or_initialize_by(user: current_user)
117 
118   if signup.persisted?
119     redirect_to @activity, notice: "You’re already signed up."
120     return
121   end
122 
123   if @activity.capacity.present? && @activity.activity_signups.count >= @activity.capacity
124     redirect_to @activity, alert: "This activity is full."
125     return
126   end
127 
128   begin
129     if signup.save
130       redirect_to @activity, notice: "You joined this activity."
131     else
132       redirect_to @activity, alert: signup.errors.full_messages.to_sentence
133     end
134   rescue ActiveRecord::RecordNotUnique
135     redirect_to @activity, alert: "This activity is full."
136   end
137 end
